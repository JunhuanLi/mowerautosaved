1function ellipse_t = fit_ellipse( x,y )
  2%#codegen
  3single status;
  4% initialize
  5orientation_tolerance = 1e-3;
  6
  7% prepare vectors, must be column vectors
  8x = x(:);
  9y = y(:);
 10
 11% remove bias of the ellipse - to make matrix inversion more accurate. (will be added later on).
 12mean_x = mean(x);
 13mean_y = mean(y);
 14x = x-mean_x;
 15y = y-mean_y;
 16
 17% the estimation for the conic equation of the ellipse
 18X = [x.^2, x.*y, y.^2, x, y ];
 19a = sum(X)/(X'*X);
 20
 21% extract parameters from the conic equation
 22[a,b,c,d,e] = deal( a(1),a(2),a(3),a(4),a(5) );
 23
 24% remove the orientation from the ellipse
 25if ( min(abs(b/a),abs(b/c)) > orientation_tolerance )
 26    
 27    orientation_rad = 1/2 * atan( b/(c-a) );
 28    cos_phi = cos( orientation_rad );
 29    sin_phi = sin( orientation_rad );
 30    [a,~,c,d,e] = deal(...
 31        a*cos_phi^2 - b*cos_phi*sin_phi + c*sin_phi^2,...
 32        0,...
 33        a*sin_phi^2 + b*cos_phi*sin_phi + c*cos_phi^2,...
 34        d*cos_phi - e*sin_phi,...
 35        d*sin_phi + e*cos_phi );
 36    [mean_x,mean_y] = deal( ...
 37        cos_phi*mean_x - sin_phi*mean_y,...
 38        sin_phi*mean_x + cos_phi*mean_y );
 39else
 40    orientation_rad = 0;
 41    cos_phi = cos( orientation_rad );
 42    sin_phi = sin( orientation_rad );
 43end
 44
 45% check if conic equation represents an ellipse
 46test = a*c;
 47% switch (1)
 48% case (test>0),  status = 1;%ellipse fount;
 49% case (test==0), status = 0;%'Parabola found';  
 50% case (test<0),  status = -1;%'Hyperbola found'; 
 51% end
 52if(test>0)
 53    status = 1;
 54else
 55    status = -1;
 56end
 57
 58% if we found an ellipse return it's data
 59if (test>0)
 60    
 61    % make sure coefficients are positive as required
 62    if (a<0), [a,c,d,e] = deal( -a,-c,-d,-e ); end
 63    
 64    % final ellipse parameters
 65    X0          = mean_x - d/2/a;
 66    Y0          = mean_y - e/2/c;
 67    F           = 1 + (d^2)/(4*a) + (e^2)/(4*c);
 68    [a,b]       = deal( sqrt( F/a ),sqrt( F/c ) );    
 69    long_axis   = 2*max(a,b);
 70    short_axis  = 2*min(a,b);
 71
 72    % rotate the axes backwards to find the center point of the original TILTED ellipse
 73    R           = [ cos_phi sin_phi; -sin_phi cos_phi ];
 74    P_in        = R * [X0;Y0];
 75    X0_in       = P_in(1);
 76    Y0_in       = P_in(2);
 77    
 78    % pack ellipse into a structure
 79    ellipse_t = struct( ...
 80        'a',a,...
 81        'b',b,...
 82        'phi',orientation_rad,...
 83        'X0',X0,...
 84        'Y0',Y0,...
 85        'X0_in',X0_in,...
 86        'Y0_in',Y0_in,...
 87        'long_axis',long_axis,...
 88        'short_axis',short_axis,...
 89        'status',status );
 90else
 91    % report an empty structure
 92    ellipse_t = struct( ...
 93        'a',0,...
 94        'b',0,...
 95        'phi',0,...
 96        'X0',0,...
 97        'Y0',0,...
 98        'X0_in',0,...
 99        'Y0_in',0,...
100        'long_axis',0,...
101        'short_axis',0,...
102        'status',status );
103end
104
105end
